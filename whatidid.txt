buildMessage :: MessageLevel -> Text -> IO Text
buildMessage lvl msg = do
    now <- {-# SCC "getCurrentTime" #-} getCurrentTime
    let showingNow = {-# SCC "showingNow" #-} (T.pack $ show now)
        showingLvl = {-# SCC "showingLvl" #-} (T.pack $ show lvl)
        doSplotch = {-# SCC "doSplotch" #-} (splotch lvl)
        msg' = {-# SCC "ccccooombbine" #-} showingNow `T.append` " || [" `T.append` showingLvl `T.append` "]" `T.append` doSplotch `T.append` msg
    return $ msg'
